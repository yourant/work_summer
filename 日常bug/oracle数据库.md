env.engine.execute(f'alter table \"{no_table.name.upper()}\" drop primary key')

    f'alter table \"{no_table.name.upper()}\" ADD ID INTEGER GENERATED BY DEFAULT AS IDENTITY (START WITH 1 INCREMENT BY 1 )')
